% ------------------------------------------------------------
% HEADER CONTENT as a macro, not a savebox
% Built fresh each page inside \LTR — bidi stays out
% ------------------------------------------------------------
\newcommand{\docheadercontent}{%
  \LTR{%
    \begin{tabularx}{\textwidth}{@{}lXr@{}}
      % ROW 1: logoleft | title | logoright
      \ifdefined\logoleft
        \includegraphics[height=\logoheight]{\logoleft}%
      \fi
      &
      \centering
      \ifdefined\title
        {\large\bfseries\title}%
      \fi
      &
      \ifdefined\logoright
        \includegraphics[height=\logoheight]{\logoright}%
      \fi
      \\[4pt]
      % ROW 2: author | date | meta
      {\headersmall\ifdefined\author\author\fi}
      &
      \centering{\headersmall\ifdefined\date\date\fi}
      &
      \raggedleft{\headersmall\ifdefined\meta\meta\fi}
      \\
      \midrule
    \end{tabularx}%
  }%
}
